26 inline void _sfpu_load_config32_(const uint dest, const uint upper16, const uint lower16)
27 {
28  // registers 11 through 14 are programmable "constants" which are shared across all 4 rows
29  // They are updated only through the CONFIG path, which uses LREG[0] first and then copies it to the desired register location
30  TTI_SFPLOADI(0, 10, lower16); // insmod == A will write the lower bits, and not affect the upper bits;
31  TTI_SFPLOADI(0, 8, upper16); // insmod == 8 will write the upper bits, and not affect the lower bits;
32  TTI_SFPCONFIG(0, dest, 0);
33 }
